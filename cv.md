function findUniq(arr) {
  let unique = 0;
  let matches = arr.filter((num) => num == arr[0]);
  if (matches.length === 1) {
    unique = arr[0];
  } else {
    unique = arr.find((num) => num !== arr[0]);
  }
  return unique;
}
